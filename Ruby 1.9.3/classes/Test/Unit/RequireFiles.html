def non_options(files, options)
  return false if !super
  result = false
  files.each {|f|
    d = File.dirname(path = File.expand_path(f))
    unless $:.include? d
      $: << d
    end
    begin
      require path unless options[:parallel]
      result = true
    rescue LoadError
      puts "#{f}: #{$!}"
    end
  }
  result
end
