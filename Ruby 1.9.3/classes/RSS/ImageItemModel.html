def self.append_features(klass)
  super

  klass.install_have_child_element("item", IMAGE_URI, "?",
                                   "#{IMAGE_PREFIX}_item")
  klass.install_must_call_validator(IMAGE_PREFIX, IMAGE_URI)
end
