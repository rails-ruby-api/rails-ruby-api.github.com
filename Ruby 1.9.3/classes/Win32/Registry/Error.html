def initialize(code)
  @code = code
  msg = "\00"".force_encoding(Encoding::ASCII_8BIT) * 1024
  len = FormatMessageA.call(0x1200, 0, code, 0, msg, 1024, 0)
  msg = msg[0, len].force_encoding(Encoding.find(Encoding.locale_charmap))
  super msg.tr("\r", '').chomp
end
