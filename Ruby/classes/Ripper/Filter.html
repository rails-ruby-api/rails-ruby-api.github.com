def parse(init = nil)
  data = init
  @__lexer.lex.each do |pos, event, tok|
    @__line, @__col = *pos
    data = if respond_to?(event, true)
           then __send__(event, tok, data)
           else on_default(event, tok, data)
           end
  end
  data
end
