def normalize
  parts = []

  newlines = 0

  @parts.each do |part|
    case part
    when %r^\s*\n/ then
      newlines += 1
      parts << part if newlines == 1
    else
      newlines = 0
      parts << part
    end
  end

  parts.pop if parts.last =~ %r\A\r?\n\z/

  @parts = parts
end
