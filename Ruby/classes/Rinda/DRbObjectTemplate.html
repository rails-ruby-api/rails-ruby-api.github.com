def ===(ro)
  return true if super(ro)
  unless @drb_uri.nil?
    return false unless (@drb_uri === ro.__drburi rescue false)
  end
  unless @drb_ref.nil?
    return false unless (@drb_ref === ro.__drbref rescue false)
  end
  true
end
