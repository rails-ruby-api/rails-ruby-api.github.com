def self.select_message(receiver, message, candidates, sep = ".")
  candidates.grep(%r^#{message}/).collect do |e|
    case e
    when %r^[a-zA-Z_]/
      receiver + sep + e
    when %r^[0-9]/
    when *Operators
      #receiver + " " + e
    end
  end
end
