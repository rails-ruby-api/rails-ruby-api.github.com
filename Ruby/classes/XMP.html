  def initialize(bind = nil)

    bind = IRB::Frame.top(1) unless bind
    ws = IRB::WorkSpace.new(bind)

def puts(exps)
  @io.puts exps

  if @irb.context.ignore_sigint
    begin
      trap_proc_b = trap("SIGINT"){@irb.signal_handle}
      catch(:IRB_EXIT) do
        @irb.eval_input
      end
    ensure
      trap("SIGINT", trap_proc_b)
    end
  else
    catch(:IRB_EXIT) do
      @irb.eval_input
    end
  end
end
