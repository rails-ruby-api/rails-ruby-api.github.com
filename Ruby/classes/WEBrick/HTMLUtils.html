def escape(string)
  str = string ? string.dup : ""
  str.gsub!(%r&/, '&amp;')
  str.gsub!(%r\"/, '&quot;')
  str.gsub!(%r>/, '&gt;')
  str.gsub!(%r</, '&lt;')
  str
end
