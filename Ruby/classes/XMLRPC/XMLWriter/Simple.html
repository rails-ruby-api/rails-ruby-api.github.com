def element(name, attrs, *children)
  raise "attributes not yet implemented" unless attrs.nil?
  if children.empty?
    "<#{name}/>"
  else
    "<#{name}>" + children.join("") + "</#{name}>"
  end
end

def text(txt)
  cleaned = txt.dup
  cleaned.gsub!(%r&/, '&amp;')
  cleaned.gsub!(%r</, '&lt;')
  cleaned.gsub!(%r>/, '&gt;')
  cleaned
end
