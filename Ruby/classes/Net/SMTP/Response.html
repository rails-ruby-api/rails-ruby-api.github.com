Source: show

def capabilities
  return {} unless @string[3, 1] == '-'
  h = {}
  @string.lines.drop(1).each do |line|
    k, *v = line[4..-1].chomp.split
    h[k] = v
  end
  h
end

def exception_class
  case @status
  when %r\A4/  then SMTPServerBusy
  when %r\A50/ then SMTPSyntaxError
  when %r\A53/ then SMTPAuthenticationError
  when %r\A5/  then SMTPFatalError
  else             SMTPUnknownError
  end
end
