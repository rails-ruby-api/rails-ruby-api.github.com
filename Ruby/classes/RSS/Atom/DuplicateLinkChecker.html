def validate_duplicate_links(links)
  link_infos = {}
  links.each do |link|
    rel = link.rel || "alternate"
    next unless rel == "alternate"
    key = [link.hreflang, link.type]
    if link_infos.has_key?(key)
      raise TooMuchTagError.new("link", tag_name)
    end
    link_infos[key] = true
  end
end
