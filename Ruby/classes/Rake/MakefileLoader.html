def load(fn)
  lines = File.read fn
  lines.gsub!(%r\\ /, SPACE_MARK)
  lines.gsub!(%r#[^\n]*\n/, "")
  lines.gsub!(%r\\\n/, ' ')
  lines.each_line do |line|
    process_line(line)
  end
end
