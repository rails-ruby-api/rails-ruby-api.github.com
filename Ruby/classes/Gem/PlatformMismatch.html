def wordy
  prefix = "Found #{@name} (#{@version})"

  if @platforms.size == 1
    "#{prefix}, but was for platform #{@platforms[0]}"
  else
    "#{prefix}, but was for platforms #{@platforms.join(' ,')}"
  end
end
