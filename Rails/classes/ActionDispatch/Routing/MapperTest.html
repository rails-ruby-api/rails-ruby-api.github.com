def test_map_wildcard
  fakeset = FakeSet.new
  mapper = Mapper.new fakeset
  mapper.match '/*path', :to => 'pages#show'
  assert_equal '/*path(.:format)', fakeset.conditions.first[:path_info]
  assert_equal(%r.+?/, fakeset.requirements.first[:path])
end
