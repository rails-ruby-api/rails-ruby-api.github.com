def test_decrement
  @cache.write('foo', 3, :raw => true)
  assert_equal 3, @cache.read('foo').to_i
  assert_equal 2, @cache.decrement('foo')
  assert_equal 2, @cache.read('foo').to_i
  assert_equal 1, @cache.decrement('foo')
  assert_equal 1, @cache.read('foo').to_i
end
