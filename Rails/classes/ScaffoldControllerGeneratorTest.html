def test_controller_skeleton_is_created
  run_generator

  assert_file "app/controllers/users_controller.rb" do |content|
    assert_match(%rclass UsersController < ApplicationController/, content)

    assert_instance_method :index, content do |m|
      assert_match(%r@users = User\.all/, m)
    end

    assert_instance_method :show, content do |m|
      assert_match(%r@user = User\.find\(params\[:id\]\)/, m)
    end

    assert_instance_method :new, content do |m|
      assert_match(%r@user = User\.new/, m)
    end

    assert_instance_method :edit, content do |m|
      assert_match(%r@user = User\.find\(params\[:id\]\)/, m)
    end

    assert_instance_method :create, content do |m|
      assert_match(%r@user = User\.new\(params\[:user\]\)/, m)
      assert_match(%r@user\.save/, m)
      assert_match(%r@user\.errors/, m)
    end

    assert_instance_method :update, content do |m|
      assert_match(%r@user = User\.find\(params\[:id\]\)/, m)
      assert_match(%r@user\.update_attributes\(params\[:user\]\)/, m)
      assert_match(%r@user\.errors/, m)
    end

    assert_instance_method :destroy, content do |m|
      assert_match(%r@user = User\.find\(params\[:id\]\)/, m)
      assert_match(%r@user\.destroy/, m)
    end
  end
end

def test_customized_orm_is_used
  klass = Class.new(Rails::Generators::ActiveModel) do
    def self.all(klass)
      "#{klass}.find(:all)"
    end
  end

  Unknown::Generators.const_set(:ActiveModel, klass)
  run_generator ["User", "--orm=unknown"]

  assert_file "app/controllers/users_controller.rb" do |content|
    assert_match(%rclass UsersController < ApplicationController/, content)

    assert_instance_method :index, content do |m|
      assert_match(%r@users = User\.find\(:all\)/, m)
      assert_no_match(%r@users = User\.all/, m)
    end
  end
ensure
  Unknown::Generators.send :remove_const, :ActiveModel
end

def test_views_are_generated
  run_generator

  %w(
    index
    edit
    new
    show
  ).each { |view| assert_file "app/views/users/#{view}.html.erb" }
  assert_no_file "app/views/layouts/users.html.erb"
end
