def mail_with_i18n_subject(recipient)
  @recipient  = recipient
  I18n.locale = :de
  mail(:to => recipient, :subject => "#{I18n.t :email_subject} #{recipient}",
    :from => "system@loudthinking.com", :date => Time.local(2004, 12, 12))
end
