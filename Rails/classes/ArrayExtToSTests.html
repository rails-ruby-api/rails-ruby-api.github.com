def test_to_s_db
  collection = [
    Class.new { def id() 1 end }.new,
    Class.new { def id() 2 end }.new,
    Class.new { def id() 3 end }.new
  ]

  assert_equal "null", [].to_s(:db)
  assert_equal "1,2,3", collection.to_s(:db)
end
