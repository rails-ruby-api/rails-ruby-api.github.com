def rake(*tasks)
  tasks.each do |task|
    cmd = "bundle exec rake #{task}"
    puts "Running command: #{cmd}"
    return false unless system(cmd)
  end
  true
end

def run!(options = {})
  self.options.update(options)
  Dir.chdir(dir) do
    announce(heading)
    ENV['IM'] = identity_map?.inspect
    rake(*tasks)
  end
end

def tasks
