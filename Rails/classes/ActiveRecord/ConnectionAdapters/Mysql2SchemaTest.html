def setup
  @connection = ActiveRecord::Base.connection
  db          = Post.connection_pool.spec.config[:database]
  table       = Post.table_name
  @db_name    = db

  @omgpost = Class.new(ActiveRecord::Base) do
    set_table_name "#{db}.#{table}"
    def self.name; 'Post'; end
  end
end
