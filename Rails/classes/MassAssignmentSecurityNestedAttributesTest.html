def nested_attributes_hash(association, collection = false, except = [:id])
  if collection
    { :first_name => 'David' }.merge(:"#{association}_attributes" => [attributes_hash.except(*except)])
  else
    { :first_name => 'David' }.merge(:"#{association}_attributes" => attributes_hash.except(*except))
  end
end
