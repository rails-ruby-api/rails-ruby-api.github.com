def assert_admin_attributes(person, create = false)
  unless create
    assert_nil person.id
  else
    assert !!person.id
  end
  assert_equal 'Josh', person.first_name
  assert_equal 'm',    person.gender
  assert_equal 'rides a sweet bike', person.comments
end
