def assert_utf8
  if RUBY_VERSION < '1.9'
    assert_equal "UTF8", $KCODE
  else
    assert_equal Encoding::UTF_8, Encoding.default_external
    assert_equal Encoding::UTF_8, Encoding.default_internal
  end
end
