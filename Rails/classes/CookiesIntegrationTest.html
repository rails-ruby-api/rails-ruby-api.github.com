def test_setting_cookies_raises_after_stream_back_to_client
  with_test_route_set do
    get '/set_cookies'
    assert_raise(ActionDispatch::ClosedError) {
      request.cookie_jar['alert'] = 'alert'
      cookies['alert'] = 'alert'
    }
  end
end
