def test_duplicable
  NO.each do |v|
    assert !v.duplicable?
    begin
      v.dup
      fail
    rescue Exception
    end
  end

  YES.each do |v|
    assert v.duplicable?
    assert_nothing_raised { v.dup }
  end
end
