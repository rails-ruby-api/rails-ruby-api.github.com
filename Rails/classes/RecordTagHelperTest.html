def test_block_not_in_erb_multiple_calls
  expected = %Q(<div class="post bar" id="post_45">#{@post.body}</div>)
  actual = div_for(@post, :class => "bar") { @post.body }
  assert_dom_equal expected, actual
  actual = div_for(@post, :class => "bar") { @post.body }
  assert_dom_equal expected, actual
end
