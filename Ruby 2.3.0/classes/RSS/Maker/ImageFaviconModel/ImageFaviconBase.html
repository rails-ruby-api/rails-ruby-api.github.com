def to_feed(feed, current)
  if current.respond_to?(:image_favicon=) and have_required_values?
    favicon = current.class::ImageFavicon.new
    setup_values(favicon)
    setup_other_elements(favicon)
    current.image_favicon = favicon
  end
end
