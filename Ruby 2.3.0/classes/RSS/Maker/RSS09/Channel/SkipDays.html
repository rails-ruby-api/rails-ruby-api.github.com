def to_feed(rss, channel)
  unless @days.empty?
    skipDays = Rss::Channel::SkipDays.new
    channel.skipDays = skipDays
    set_parent(skipDays, channel)
    @days.each do |day|
      day.to_feed(rss, skipDays.days)
    end
  end
end
