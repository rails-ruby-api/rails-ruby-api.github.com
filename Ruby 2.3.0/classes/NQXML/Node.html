def nodeType
  if @entity.instance_of? NQXML::Text then :TEXT
  elsif @entity.instance_of? NQXML::Comment then :COMMENT
  #elsif @entity.instance_of? NQXML::Element then :ELEMENT
  elsif @entity.instance_of? NQXML::Tag then :ELEMENT
  else :ELSE
  end
end
