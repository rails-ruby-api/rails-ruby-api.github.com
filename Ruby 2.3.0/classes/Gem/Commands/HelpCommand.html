def execute
  arg = options[:args][0]

  _, help = SUBCOMMANDS.find do |command,|
    begins? command, arg
  end

  if help then
    if Symbol === help then
      send help
    else
      say help
    end
    return
  end

  if options[:help] then
    show_help

  elsif arg then
    show_command_help arg

  else
    say Gem::Command::HELP
  end
end
