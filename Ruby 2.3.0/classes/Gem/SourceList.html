Class Gem::SourceList < Object

  • lib/rubygems/source_list.rb

The SourceList represents the sources rubygems has been configured to use. A source may be created from an array of sources:

Gem::SourceList.from %w[https://rubygems.example https://internal.example]

Or by adding them:

sources = Gem::SourceList.new
sources.add 'https://rubygems.example'

The most common way to get a SourceList is Gem.sources.

def <<(obj)
  src = case obj
        when URI
          Gem::Source.new(obj)
        when Gem::Source
          obj
        else
          Gem::Source.new(URI.parse(obj))
        end

  @sources << src
  src
end

def delete source
  if source.kind_of? Gem::Source
    @sources.delete source
  else
    @sources.delete_if { |x| x.uri.to_s == source.to_s }
  end
end

def include?(other)
  if other.kind_of? Gem::Source
    @sources.include? other
  else
    @sources.find { |x| x.uri.to_s == other.to_s }
  end
end
