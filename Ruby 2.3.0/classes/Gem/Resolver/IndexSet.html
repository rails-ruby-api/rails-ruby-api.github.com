def find_all req
  res = []

  return res unless @remote

  name = req.dependency.name

  @all[name].each do |uri, n|
    if req.match? n, @prerelease then
      res << Gem::Resolver::IndexSpecification.new(
        self, n.name, n.version, uri, n.platform)
    end
  end

  res
end
