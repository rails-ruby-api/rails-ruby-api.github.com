def Daemon.start
  exit!(0) if fork
  Process::setsid
  exit!(0) if fork
  Dir::chdir("/")
  File::umask(0)
  STDIN.reopen(IO::NULL)
  STDOUT.reopen(IO::NULL, "w")
  STDERR.reopen(IO::NULL, "w")
  yield if block_given?
end
