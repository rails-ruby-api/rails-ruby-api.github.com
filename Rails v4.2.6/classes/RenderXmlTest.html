def test_rendering_with_object_location_should_set_header_with_url_for
  with_routing do |set|
    set.draw do
      resources :customers
      get ':controller/:action'
    end

    get :render_with_object_location
    assert_equal "http://www.nextangle.com/customers/1", @response.headers["Location"]
  end
end
