def silence(temporary_level = Logger::ERROR)
  if silencer
    begin
      old_local_level            = local_levels[local_log_id]
      local_levels[local_log_id] = temporary_level

      yield self
    ensure
      if old_local_level
        local_levels[local_log_id] = old_local_level
      else
        local_levels.delete(local_log_id)
      end
    end
  else
    yield self
  end
end
