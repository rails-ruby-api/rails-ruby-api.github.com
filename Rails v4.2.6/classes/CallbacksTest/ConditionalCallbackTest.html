def test_save_conditional_person
  person = ConditionalPerson.new
  person.save
  assert_equal [
    [:before_save, :proc],
    [:before_save, :proc],
    [:before_save, :symbol],
    [:before_save, :symbol],
    [:before_save, :string],
    [:before_save, :string],
    [:before_save, :combined_symbol],
  ], person.history
end
