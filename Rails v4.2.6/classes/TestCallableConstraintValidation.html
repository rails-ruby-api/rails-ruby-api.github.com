def test_constraint_with_object_not_callable

    ActionDispatch::Routing::RouteSet.new.draw do
      ok = lambda { |env| [200, { 'Content-Type' => 'text/plain' }, []] }
      get '/test', to: ok, constraints: Object.new
    end
  end
