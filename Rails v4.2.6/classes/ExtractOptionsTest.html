def test_extract_options
  assert_equal({}, [].extract_options!)
  assert_equal({}, [1].extract_options!)
  assert_equal({ a: :b }, [{ a: :b }].extract_options!)
  assert_equal({ a: :b }, [1, { a: :b }].extract_options!)
end
