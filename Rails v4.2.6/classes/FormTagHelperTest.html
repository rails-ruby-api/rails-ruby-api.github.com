def form_text(action = "http://www.example.com", options = {})
  remote, enctype, html_class, id, method = options.values_at(:remote, :enctype, :html_class, :id, :method)

  method = method.to_s == "get" ? "get" : "post"

  txt =  %Q{<form accept-charset="UTF-8" action="#{action}"}
  txt << %Q{ enctype="multipart/form-data"} if enctype
  txt << %Q{ data-remote="true"} if remote
  txt << %Q{ class="#{html_class}"} if html_class
  txt << %Q{ id="#{id}"} if id
  txt << %Q{ method="#{method}">}
end

def hidden_fields(options = {})
  method = options[:method]
  enforce_utf8 = options.fetch(:enforce_utf8, true)

  ''.tap do |txt|
    if enforce_utf8
      txt << %Q{<input name="utf8" type="hidden" value="&#x2713;" />}
    end

    if method && !%w(get post).include?(method.to_s)
      txt << %Q{<input name="_method" type="hidden" value="#{method}" />}
    end
  end
end

def test_field_set_tag_in_erb
  output_buffer = render_erb("<%= field_set_tag('Your details') do %>Hello world!<% end %>")

  expected = %Q(<fieldset><legend>Your details</legend>Hello world!</fieldset>)
  assert_dom_equal expected, output_buffer

  output_buffer = render_erb("<%= field_set_tag do %>Hello world!<% end %>")

  expected = %Q(<fieldset>Hello world!</fieldset>)
  assert_dom_equal expected, output_buffer

  output_buffer = render_erb("<%= field_set_tag('') do %>Hello world!<% end %>")

  expected = %Q(<fieldset>Hello world!</fieldset>)
  assert_dom_equal expected, output_buffer

  output_buffer = render_erb("<%= field_set_tag('', :class => 'format') do %>Hello world!<% end %>")

  expected = %Q(<fieldset class="format">Hello world!</fieldset>)
  assert_dom_equal expected, output_buffer

  output_buffer = render_erb("<%= field_set_tag %>")

  expected = %Q(<fieldset></fieldset>)
  assert_dom_equal expected, output_buffer

  output_buffer = render_erb("<%= field_set_tag('You legend!') %>")

  expected = %Q(<fieldset><legend>You legend!</legend></fieldset>)
  assert_dom_equal expected, output_buffer
end
