def test_structure_load
  filename = "awesome-file.sql"
  Kernel.expects(:system).with('mysql', '--execute', %Q{SET FOREIGN_KEY_CHECKS = 0; SOURCE #{filename}; SET FOREIGN_KEY_CHECKS = 1}, "--database", "test-db")
    .returns(true)

  ActiveRecord::Tasks::DatabaseTasks.structure_load(@configuration, filename)
end
