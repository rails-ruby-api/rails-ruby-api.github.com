def test_url_sub_key_merges_correctly
  hash = {"url" => 'abstract://foo?encoding=utf8&', "adapter" => "sqlite3", "host" => "bar", "pool" => "3"}
  spec = resolve :production, 'production' => hash
  assert_equal({
    "adapter"  => "abstract",
    "host"     => "foo",
    "encoding" => "utf8",
    "pool"     => "3" }, spec)
end
