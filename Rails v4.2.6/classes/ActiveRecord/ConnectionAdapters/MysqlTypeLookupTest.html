def test_string_types
  assert_lookup_type :string, "enum('one', 'two', 'three')"
  assert_lookup_type :string, "ENUM('one', 'two', 'three')"
  assert_lookup_type :string, "set('one', 'two', 'three')"
  assert_lookup_type :string, "SET('one', 'two', 'three')"
end
