def test_cache_is_per_pid

  cache = StatementPool.new nil, 10
  cache['foo'] = 'bar'
  assert_equal 'bar', cache['foo']

  pid = fork {
    lookup = cache['foo'];
    exit!(!lookup)
  }

  Process.waitpid pid
  assert $?.success?, 'process should exit successfully'
end
