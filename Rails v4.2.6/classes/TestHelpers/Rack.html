def app(env = "production")
  old_env = ENV["RAILS_ENV"]
  @app ||= begin
    ENV["RAILS_ENV"] = env
    require "#{app_path}/config/environment"
    Rails.application
  end
ensure
  ENV["RAILS_ENV"] = old_env
end
