def setup
  super
  s = self
  routes = ActionDispatch::Routing::RouteSet.new
  routes.append do
    get '/hello'   => s.simple_app('fail')
    get '/goodbye' => s.simple_app('goodbye')
  end

  routes.draw do
    get '/hello' => s.simple_app('hello')
  end
  @app = self.class.build_app routes
end
