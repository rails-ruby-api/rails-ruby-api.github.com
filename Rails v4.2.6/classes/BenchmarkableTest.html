def test_outside_level
  logger.level = ActiveSupport::Logger::ERROR
  benchmark('skipped_debug_run', :level => :debug) { }
  assert_no_match(/skipped_debug_run/, buffer.last)
ensure
  logger.level = ActiveSupport::Logger::DEBUG
end
