test_multiparameter_attributes_on_time_with_skip_time_zone_conversion_for_attributes() Link

# File activerecord/test/cases/multiparameter_attributes_test.rb, line 227
def test_multiparameter_attributes_on_time_with_skip_time_zone_conversion_for_attributes
  with_timezone_config default: :utc, aware_attributes: true, zone: -28800 do
    Topic.skip_time_zone_conversion_for_attributes = [:written_on]
    attributes = {
      "written_on(1i)" => "2004", "written_on(2i)" => "6", "written_on(3i)" => "24",
      "written_on(4i)" => "16", "written_on(5i)" => "24", "written_on(6i)" => "00"
    }
    topic = Topic.find(1)
    topic.attributes = attributes
    assert_equal Time.utc(2004, 6, 24, 16, 24, 0), topic.written_on
    assert_equal false, topic.written_on.respond_to?(:time_zone)
  end
ensure
  Topic.skip_time_zone_conversion_for_attributes = []
end
