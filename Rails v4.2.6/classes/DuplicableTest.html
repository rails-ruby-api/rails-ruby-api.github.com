def test_duplicable
  RAISE_DUP.each do |v|
    assert !v.duplicable?
    assert_raises(TypeError, v.class.name) { v.dup }
  end

  ALLOW_DUP.each do |v|
    assert v.duplicable?, "#{ v.class } should be duplicable"
    assert_nothing_raised { v.dup }
  end
end
