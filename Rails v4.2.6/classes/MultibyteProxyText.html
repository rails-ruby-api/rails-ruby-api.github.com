def with_custom_encoder(encoder)
  original_proxy_class = ActiveSupport::Multibyte.proxy_class

  begin
    ActiveSupport::Multibyte.proxy_class = encoder

    yield
  ensure
    ActiveSupport::Multibyte.proxy_class = original_proxy_class
  end
end
