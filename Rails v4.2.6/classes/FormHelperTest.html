  assert_dom_equal(
    %Q{<label for="post_#{pid}_title">Title</label>},
    label("post[]", "title")
  )
  assert_dom_equal(
    %Q{<input id="post_#{pid}_title" name="post[#{pid}][title]" type="text" value="Hello World" />},
    text_field("post[]","title")
  )
  assert_dom_equal(
    %Q{<textarea id="post_#{pid}_body" name="post[#{pid}][body]">\nBack to the hill and over it again!</textarea>},
    text_area("post[]", "body")
  )
  assert_dom_equal(
    %Q{<input name="post[#{pid}][secret]" type="hidden" value="0" /><input checked="checked" id="post_#{pid}_secret" name="post[#{pid}][secret]" type="checkbox" value="1" />},
    check_box("post[]", "secret")
  )
 assert_dom_equal(
    %Q{<input checked="checked" id="post_#{pid}_title_hello_world" name="post[#{pid}][title]" type="radio" value="Hello World" />},
    radio_button("post[]", "title", "Hello World")
  )
  assert_dom_equal(
    %Q{<input id="post_#{pid}_title_goodbye_world" name="post[#{pid}][title]" type="radio" value="Goodbye World" />},

def hidden_fields(options = {})
  method = options[:method]

  if options.fetch(:enforce_utf8, true)
    txt = %Q{<input name="utf8" type="hidden" value="&#x2713;" />}
  else
    txt = ''
  end

  if method && !%w(get post).include?(method.to_s)
    txt << %Q{<input name="_method" type="hidden" value="#{method}" />}
  end

  txt
end
