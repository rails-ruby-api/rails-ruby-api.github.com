def edit
  @workshop = Workshop.new(params[:workshop_id])
  @session = Session.new(params[:id])
  @url = [@workshop, @session, format: params[:format]]
  render inline: "<%= url_for(@url) %>\n<%= link_to('Session', @url) %>"
end
