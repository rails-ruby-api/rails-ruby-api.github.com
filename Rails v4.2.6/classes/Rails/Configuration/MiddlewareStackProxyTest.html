def test_order
  @stack.swap :foo
  @stack.delete :foo

  mock = Minitest::Mock.new
  mock.expect :send, nil, [:swap, :foo]
  mock.expect :send, nil, [:delete, :foo]

  @stack.merge_into mock
  mock.verify
end
