def test_force_ssl_redirects_to_same_format
  with_routing do |set|
    set.draw do
      get '/foo', :to => 'force_ssl_controller_level#banana'
    end

    get :banana, :format => :json
    assert_response 301
    assert_equal 'https://test.host/foo.json', redirect_to_url
  end
end
