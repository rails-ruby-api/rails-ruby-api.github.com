def test_with_hook
  add_to_config <<-RUBY
    runner do |app|
      app.config.ran = true
    end
  RUBY

  assert_match "true", Dir.chdir(app_path) { %x`bundle exec rails runner "puts Rails.application.config.ran"` }
end
