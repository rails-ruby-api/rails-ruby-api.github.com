def test_root_path
  app('development')

  controller :foo, <<-RUBY
    class FooController < ApplicationController
      def index
        render :text => "foo"
      end
    end
  RUBY

  app_file 'config/routes.rb', <<-RUBY
    Rails.application.routes.draw do
      get 'foo', :to => 'foo#index'
      root :to => 'foo#index'
    end
  RUBY

  remove_file 'public/index.html'

  get '/'
  assert_equal 'foo', last_response.body
end
