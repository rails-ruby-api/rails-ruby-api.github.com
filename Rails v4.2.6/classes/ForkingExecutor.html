def shutdown
  pool = @size.times.map {
    fork {
      DRb.stop_service
      queue = DRbObject.new_with_uri @url
      while job = queue.pop
        klass    = job[0]
        method   = job[1]
        reporter = job[2]
        result = Minitest.run_one_method klass, method
        if result.error?
          translate_exceptions result
        end
        queue.record reporter, result
      end
    }
  }
  @size.times { @queue << nil }
  pool.each { |pid| Process.waitpid pid }
end
