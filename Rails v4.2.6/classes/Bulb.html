def self.new(attributes = {}, &block)
  bulb_type = (attributes || {}).delete(:bulb_type)

  if bulb_type.present?
    bulb_class = "#{bulb_type.to_s.camelize}Bulb".constantize
    bulb_class.new(attributes, &block)
  else
    super
  end
end
