def self.dump insures
  numbers = insures.map { |insure| INSURES.index(insure) }
  numbers.inject(0) { |sum, n| sum + (1 << n) }
end

def self.load mask
  INSURES.select do |insure|
    (1 << INSURES.index(insure)) & mask.to_i > 0
  end
end
