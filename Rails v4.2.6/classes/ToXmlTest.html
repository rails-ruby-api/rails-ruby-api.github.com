def test_to_xml_with_block
  xml = [
    { name: "David", age: 26, age_in_millis: 820497600000 },
    { name: "Jason", age: 31, age_in_millis: BigDecimal.new('1.0') }
  ].to_xml(skip_instruct: true, indent: 0) do |builder|
    builder.count 2
  end

  assert xml.include?(%Q(<count>2</count>)), xml
end

def test_to_xml_with_hash_elements
  xml = [
    { name: "David", age: 26, age_in_millis: 820497600000 },
    { name: "Jason", age: 31, age_in_millis: BigDecimal.new('1.0') }
  ].to_xml(skip_instruct: true, indent: 0)

  assert_equal '<objects type="array"><object>', xml.first(30)
  assert xml.include?(%Q(<age type="integer">26</age>)), xml
  assert xml.include?(%Q(<age-in-millis type="integer">820497600000</age-in-millis>)), xml
  assert xml.include?(%Q(<name>David</name>)), xml
  assert xml.include?(%Q(<age type="integer">31</age>)), xml
  assert xml.include?(%Q(<age-in-millis type="decimal">1.0</age-in-millis>)), xml
  assert xml.include?(%Q(<name>Jason</name>)), xml
end

def test_to_xml_with_indent_set

  assert xml.include?(%Q(\n        <name>David</name>))
  assert xml.include?(%Q(\n        <street-address>Evergreen</street-address>))
  assert xml.include?(%Q(\n        <name>Jason</name>))
end
