def test_failed_json_decoding
  assert_raise(ActiveSupport::JSON.parse_error) { ActiveSupport::JSON.decode(%Q(undefined)) }
  assert_raise(ActiveSupport::JSON.parse_error) { ActiveSupport::JSON.decode(%Q({a: 1})) }
  assert_raise(ActiveSupport::JSON.parse_error) { ActiveSupport::JSON.decode(%Q({: 1})) }
  assert_raise(ActiveSupport::JSON.parse_error) { ActiveSupport::JSON.decode(%Q()) }
end
