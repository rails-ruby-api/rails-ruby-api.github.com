def test_assert_response_fixnum
  @response = FakeResponse.new 400
  assert_response 400

  assert_raises(Minitest::Assertion) {
    assert_response :unauthorized
  }

  assert_raises(Minitest::Assertion) {
    assert_response 500
  }

def test_assert_response_predicate_methods

      assert_response :unauthorized
    }
  end
end
