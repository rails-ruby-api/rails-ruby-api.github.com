def test_dynamic_render_with_absolute_path
  file = Tempfile.new('name')
  file.write "secrets!"
  file.flush
  assert_raises ActionView::MissingTemplate do
    response = get :dynamic_render, { id: file.path }
  end
ensure
  file.close
  file.unlink
end
