def test_asset_url_with_config_asset_host
  @module.instance_eval do
    def config
      Struct.new(:asset_host).new("http://www.example.com")
    end
  end

  assert @module.config.asset_host
  assert_equal "http://www.example.com/foo", @module.asset_url("foo")
end
