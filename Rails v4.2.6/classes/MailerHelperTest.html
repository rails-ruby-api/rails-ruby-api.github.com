def test_use_block_format
  mail = HelperMailer.use_block_format
  expected = <<-TEXT
This is the first paragraph.

The second paragraph.

* item1
* item2
* item3
  TEXT
  assert_equal expected.gsub("\n", "\r\n"), mail.body.encoded
end
