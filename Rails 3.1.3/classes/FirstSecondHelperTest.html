def test_ordering
  mail = FirstMailer.share
  assert_equal "first mail", mail.body.to_s.strip
  mail = SecondMailer.share
  assert_equal "second mail", mail.body.to_s.strip
  mail = FirstMailer.share
  assert_equal "first mail", mail.body.to_s.strip
  mail = SecondMailer.share
  assert_equal "second mail", mail.body.to_s.strip
end
