def test_create_database_with_encoding
  assert_equal %Q(CREATE DATABASE "matt" ENCODING = 'utf8'), create_database(:matt)
  assert_equal %Q(CREATE DATABASE "aimonetti" ENCODING = 'latin1'), create_database(:aimonetti, :encoding => :latin1)
end
