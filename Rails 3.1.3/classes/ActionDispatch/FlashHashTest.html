def test_discard_one_arg
  @hash['hello'] = 'world'
  @hash['omg']   = 'world'
  @hash.discard 'hello'

def test_each
  @hash['hello'] = 'world'
  @hash['foo'] = 'bar'

  things = []
  @hash.each do |k,v|
    things << [k,v]
  end

  assert_equal([%w{ hello world }, %w{ foo bar }].sort, things.sort)
end

def test_to_hash
  @hash['foo'] = 'bar'
  assert_equal({'foo' => 'bar'}, @hash.to_hash)

  @hash.to_hash['zomg'] = 'aaron'
  assert !@hash.key?('zomg')
  assert_equal({'foo' => 'bar'}, @hash.to_hash)
end
