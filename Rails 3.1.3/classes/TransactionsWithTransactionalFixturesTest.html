def test_automatic_savepoint_in_outer_transaction
  @first = Topic.find(1)

  begin
    Topic.transaction do
      @first.approved = true
      @first.save!
      raise
    end
  rescue
    assert !@first.reload.approved?
  end
end

def test_no_automatic_savepoint_for_inner_transaction
  @first = Topic.find(1)

  Topic.transaction do
    @first.approved = true
    @first.save!

    begin
      Topic.transaction do
        @first.approved = false
        @first.save!
        raise
      end
    rescue
    end
  end

  assert !@first.reload.approved?
end
