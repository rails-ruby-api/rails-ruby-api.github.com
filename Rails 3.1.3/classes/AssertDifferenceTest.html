def test_array_of_expressions_identify_failure
  assert_difference ['@object.num', '1 + 1'] do
    @object.increment
  end
  fail 'should not get to here'
rescue Exception => e
  assert_match(/didn't change by/, e.message)
  assert_match(/expected but was/, e.message)
end
