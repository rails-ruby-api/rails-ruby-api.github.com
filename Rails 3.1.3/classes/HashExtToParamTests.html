def test_string_hash
  assert_equal '', {}.to_param
  assert_equal 'hello=world', { :hello => "world" }.to_param
  assert_equal 'hello=10', { "hello" => 10 }.to_param
  assert_equal 'hello=world&say_bye=true', {:hello => "world", "say_bye" => true}.to_param
end
