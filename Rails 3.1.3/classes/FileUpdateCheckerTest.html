def test_should_invoke_the_block_if_a_file_has_changed
  i = 0
  checker = ActiveSupport::FileUpdateChecker.new(FILES){ i += 1 }
  checker.execute_if_updated
  sleep(1)
  FileUtils.touch(FILES)
  checker.execute_if_updated
  assert_equal 2, i
end
