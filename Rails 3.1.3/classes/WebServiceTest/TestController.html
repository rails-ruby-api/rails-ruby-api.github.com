def dump_params_keys(hash = params)
  hash.keys.sort.inject("") do |s, k|
    value = hash[k]
    value = Hash === value ? "(#{dump_params_keys(value)})" : ""
    s << ", " unless s.empty?
    s << "#{k}#{value}"
  end
end
