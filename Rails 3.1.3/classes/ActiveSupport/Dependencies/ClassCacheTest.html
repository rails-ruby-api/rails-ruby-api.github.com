def test_anonymous_class_fail
  assert_raises(ArgumentError) do
    assert_deprecated do
      @cache.new Class.new
    end
  end

  assert_raises(ArgumentError) do
    x = Class.new
    @cache[x] = x
  end

  assert_raises(ArgumentError) do
    x = Class.new
    @cache.store x
  end
end
