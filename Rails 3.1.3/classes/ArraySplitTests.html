def test_split_with_edge_values
  assert_equal [[], [2, 3, 4, 5]],  [1, 2, 3, 4, 5].split(1)
  assert_equal [[1, 2, 3, 4], []],  [1, 2, 3, 4, 5].split(5)
  assert_equal [[], [2, 3, 4], []], [1, 2, 3, 4, 5].split { |i| i == 1 || i == 5 }
end
