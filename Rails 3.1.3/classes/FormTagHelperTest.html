def form_text(action = "http://www.example.com", options = {})
  remote, enctype, html_class, id, method = options.values_at(:remote, :enctype, :html_class, :id, :method)

  method = method.to_s == "get" ? "get" : "post"

  txt =  %Q{<form accept-charset="UTF-8" action="#{action}"}
  txt << %Q{ enctype="multipart/form-data"} if enctype
  txt << %Q{ data-remote="true"} if remote
  txt << %Q{ class="#{html_class}"} if html_class
  txt << %Q{ id="#{id}"} if id
  txt << %Q{ method="#{method}">}
end

def snowman(options = {})
  method = options[:method]

  txt =  %Q{<div style="margin:0;padding:0;display:inline">}
  txt << %Q{<input name="utf8" type="hidden" value="&#x2713;" />}
  if method && !method.to_s.in?(['get','post'])
    txt << %Q{<input name="_method" type="hidden" value="#{method}" />}
  end
  txt << %Q{</div>}
end

def test_boolean_options
  assert_dom_equal %Q(<input checked="checked" disabled="disabled" id="admin" name="admin" readonly="readonly" type="checkbox" value="1" />), check_box_tag("admin", 1, true, 'disabled' => true, :readonly => "yes")

  assert_dom_equal %Q(<select id="people_" multiple="multiple" name="people[]"><option>david</option></select>), select_tag("people[]", "<option>david</option>".html_safe, :multiple => true)
  assert_dom_equal %Q(<select id="people" name="people"><option>david</option></select>), select_tag("people", "<option>david</option>".html_safe, :multiple => nil)
end

  actual = radio_button_tag('ctrlname', 'apache2.2')

def whole_form(action = "http://www.example.com", options = {})
  out = form_text(action, options) + snowman(options)

  if block_given?
    out << yield << "</form>"
  end

  out
end
