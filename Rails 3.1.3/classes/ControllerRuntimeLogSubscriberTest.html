def test_runtime_reset_before_requests
  ActiveRecord::LogSubscriber.runtime += 12345
  get :zero
  wait

  assert_equal 2, @logger.logged(:info).size
  assert_match(/\(Views: [\d.]+ms \| ActiveRecord: 0.0ms\)/, @logger.logged(:info)[1])
