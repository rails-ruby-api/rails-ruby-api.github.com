def test_tampered_data_raises
  data, hash = @verifier.generate(@data).split("--")
  assert_not_verified("#{data.reverse}--#{hash}")
  assert_not_verified("#{data}--#{hash.reverse}")
  assert_not_verified("purejunk")
end
