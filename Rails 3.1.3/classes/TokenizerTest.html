def test_less_than_without_matching_greater_than
  tokenize %Q{hello <span onmouseover="gotcha"\n<b>foo</b>\nbar</span>}
  assert_next %Q{hello }
  assert_next %Q{<span onmouseover="gotcha"\n}
  assert_next %Q{<b>}
  assert_next %Q{foo}
  assert_next %Q{</b>}
  assert_next %Q{\nbar}
  assert_next %Q{</span>}
  assert_end
end
